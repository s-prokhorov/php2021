docker images - показывает списко всех установленных локально образов
docker ps - просмотр контейнеров на локальной машине
docker ps -a - просмотр всех контейнеров, включая остановленные.
docker build - сборка контейнера на основании информации, записанной в dockerfile.
docker rm <name> - удаление контейнера
docker run <name> - старт контейнера
docker run --name <label> <name> - старт контейнера с определением его имени
docker run <name> --rm - старт контейнера с автоматическим удалением после остановки.
docker stop <name> - остановка выполнения контейнера
docker volume ls - просмотр существующих томов
docker volume create <name> - создание нового тома с указанным именем
